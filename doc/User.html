  find_by name: name

def to_token_payload
  payload = {}
  # std jwt claims
  payload['sub'] = id
  payload['iat'] = Time.now.utc.to_i
  payload['iss'] = Rails.application.secrets.jwt_issuer
  # sombra claims
  payload['role'] = role
  payload['name'] = name
  payload['hash'] = Digest::SHA256.base64digest updated_at.to_s
  payload
end
